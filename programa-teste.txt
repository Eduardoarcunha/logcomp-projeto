character hero = Scanln();
monster enemy = Scanln();

int hero_health = 100;
int enemy_health = 80;
int time = 0;

Println("Battle begins between", hero, "and", enemy);

combat ; time = 0 ; while hero_health > 0 || enemy_health > 0 progress time = time + 1 {
    enemy_health = enemy_health - 10;
    Println(hero, "attacks!", enemy, "loses 10 HP");
    hero_health = hero_health - 5;
    Println(enemy, "counterattacks!", hero, "loses 5 HP");
}   // A ideia aqui é sempre printar o tempo ao fim de cada iteração!

if hero_health > 0 {
    Println(hero, "wins!");
} else {
    Println(enemy, "wins!");
}
